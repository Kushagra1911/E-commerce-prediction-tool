    <script>
        // Simple prediction generator based on input
        function generatePrediction(input) {
            const lowerInput = input.toLowerCase();

            // Handle currency conversion
            if ((lowerInput.includes('convert') || lowerInput.includes('exchange')) &&
                (lowerInput.includes('dollar') || lowerInput.includes('usd') || lowerInput.includes('$')) &&
                (lowerInput.includes('rupee') || lowerInput.includes('inr') || lowerInput.includes('₹'))) {

                // Extract number if present, default to 1
                const numberMatch = lowerInput.match(/(\d+(?:\.\d+)?)/);
                const amount = numberMatch ? parseFloat(numberMatch[1]) : 1;
                const convertedAmount = (amount * 83.12).toFixed(2); // Current USD to INR rate

                return {
                    prediction: `₹${convertedAmount}`,
                    confidence: '100%',
                    timeframe: 'Current rate',
                    insight: `${amount} USD = ₹${convertedAmount} at today's exchange rate (1 USD = ₹83.12)`
                };
            } else if (lowerInput.includes('sales') || lowerInput.includes('revenue')) {
                return {
                    prediction: '+18.5% increase',
                    confidence: '92%',
                    timeframe: 'Next 30 days',
                    insight: 'Peak season approaching with strong market indicators'
                };
            } else if (lowerInput.includes('inventory') || lowerInput.includes('stock')) {
                return {
                    prediction: 'Restock in 12 days',
                    confidence: '89%',
                    timeframe: 'Next 2 weeks',
                    insight: 'Current demand exceeds supply by 15%'
                };
            } else if (lowerInput.includes('customer') || lowerInput.includes('users')) {
                return {
                    prediction: '2.3x lifetime value',
                    confidence: '85%',
                    timeframe: 'Next 6 months',
                    insight: 'High engagement segment showing retention patterns'
                };
            } else if (lowerInput.includes('price') || lowerInput.includes('pricing')) {
                return {
                    prediction: 'Optimal at ₹3,999',
                    confidence: '94%',
                    timeframe: 'Current market',
                    insight: 'Price elasticity suggests 8% room for increase'
                };
            } else if (lowerInput.includes('trend') || lowerInput.includes('market')) {
                return {
                    prediction: 'Rising 23% monthly',
                    confidence: '88%',
                    timeframe: 'Next quarter',
                    insight: 'Market momentum building with seasonal factors'
                };
            } else {
                return {
                    prediction: '+15.2% growth',
                    confidence: '87%',
                    timeframe: 'Next 30 days',
                    insight: 'Multiple positive indicators detected'
                };
            }
        }

        // DOM elements
        const searchInput = document.getElementById('searchInput');
        const predictBtn = document.getElementById('predictBtn');
        const btnText = document.getElementById('btnText');
        const loadingSpinner = document.getElementById('loadingSpinner');
        const predictionResult = document.getElementById('predictionResult');
        const predictionValue = document.getElementById('predictionValue');
        const confidenceValue = document.getElementById('confidenceValue');
        const timeframeValue = document.getElementById('timeframeValue');
        const insightValue = document.getElementById('insightValue');

        // Handle prediction
        function handlePredict() {
            const query = searchInput.value.trim();
            if (!query) return;

            // Show loading state
            predictBtn.disabled = true;
            btnText.style.display = 'none';
            loadingSpinner.style.display = 'block';

            // Simulate API call delay
            setTimeout(() => {
                const result = generatePrediction(query);
                
                // Update result content
                predictionValue.textContent = result.prediction;
                confidenceValue.textContent = result.confidence;
                timeframeValue.textContent = result.timeframe;
                insightValue.textContent = result.insight;
                
                // Show result
                predictionResult.classList.add('show');
                
                // Reset button state
                predictBtn.disabled = false;
                btnText.style.display = 'block';
                loadingSpinner.style.display = 'none';
            }, 1500);
        }

        // Event listeners
        predictBtn.addEventListener('click', handlePredict);

        searchInput.addEventListener('keypress', (e) => {
            if (e.key === 'Enter') {
                handlePredict();
            }
        });

        // Example query buttons
        document.querySelectorAll('.example-btn').forEach(btn => {
            btn.addEventListener('click', () => {
                searchInput.value = btn.dataset.query;
            });
        });

        // Update button state based on input
        searchInput.addEventListener('input', () => {
            predictBtn.disabled = !searchInput.value.trim();
        });

        // Initial state
        predictBtn.disabled = true;
    </script>
